    \begin{center}
        \textbf{№1 Выразить одним действием следующие формулы:}
    \end{center}

    \begin{enumerate}
        \begin{spacing}{1.5}
            \item $x \wedge y \vee \overline x \wedge \overline y = x \leftrightarrow y$

            \item $\overline x \vee y = x \rightarrow y$

            \item $\overline{x \wedge y} = x \mid y$

            \item $\overline{x \vee y} = x \downarrow y$

            \item $\overline x \wedge y \vee x \wedge \overline y = x \oplus y$

            \item $(x \rightarrow y) \wedge (y \rightarrow x) = x \leftrightarrow y$
        \end{spacing}
    \end{enumerate}

    \end{center}

    \begin{center}
        \textbf{№5 Доказать равносильность следующих формул:}
    \end{center}

    \begin{enumerate}
        \begin{spacing}{1.5}
            \item $(x \wedge y) \oplus x$ и $x \wedge \overline y$\\
            Преобразуем левую часть:\\
            $(x \wedge y) \oplus x = $
            $\overline{x \wedge y} \wedge x \vee x \wedge y \wedge \overline x = $
            $(\overline x \vee \overline y) \wedge x = $
            $x \wedge \overline y$\\
            $x \wedge \overline y = x \wedge \overline y$ - ч. т. д.

            \item $x \leftrightarrow y \rightarrow x$ и $x \wedge x \vee y$\\
            Преобразуем левую часть:\\
            $x \leftrightarrow y \rightarrow x = $
            $x \leftrightarrow \overline y \vee x = $
            $x \wedge (\overline y \vee x) \vee \overline x \wedge \overline{(\overline y \vee x)} = $\\
            $x \vee x \wedge \overline y \vee \overline x \wedge ( y \wedge \overline x) = $
            $x \vee x \wedge \overline y \vee \overline x \wedge y = x \vee y$\\
            Преобразуем правую часть:\\
            $x \wedge x \vee y = x \vee y$\\
            $x \vee y = x \vee y$ - ч. т. д.

            \item $(y \downarrow x) \wedge (x \oplus y \vee \overline{x \wedge \overline y})$ и $\overline x \wedge \overline y$\\
            Преобразуем левую часть:\\
            $(y \downarrow x) \wedge (x \oplus y \vee \overline{x \wedge \overline y}) = $
            $\overline x \wedge \overline y \wedge (\overline x \wedge y \vee x \wedge \overline y \vee \overline{x \wedge \overline y}) = $
            $\overline x \wedge \overline y$\\
            $\overline x \wedge \overline y = \overline x \wedge \overline y$ - ч. т. д.

            \item $\overline{x \rightarrow y} \vee (x \mid y)$ и $\overline x \vee \overline y$\\
            Преобразуем левую часть:\\
            $\overline{x \rightarrow y} \vee (x \mid y) = $
            $\overline{\overline x \vee y} \vee \overline x \vee \overline y = $
            $x \wedge \overline y \vee \overline x \vee \overline y = \overline x \vee \overline y$\\
            $\overline x \vee \overline y = \overline x \vee \overline y$ - ч. т. д.

            \item $((x \mid y) \mid x) \mid y$ и $\overline y$\\

            $x \wedge y \vee \overline x \wedge \overline y$\\
            $x \wedge y \vee \overline x \vee \overline y = x \wedge y \vee \overline x \vee \overline y$ - ч. т. д.

            \item $(x \oplus y) \oplus (y \downarrow x)$ и $\overline x \vee \overline y$\\
